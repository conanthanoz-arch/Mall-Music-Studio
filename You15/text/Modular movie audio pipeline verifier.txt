import os
import sys
from mutagen.mp3 import MP3

# Copy your calculated TRACKLIST configuration here to test it
TRACKLIST = [
    {"title": "Late Night Patrol", "artist": "Lo-Fi Guard Collective", "duration_sec": 180.0},
    {"title": "Neon Escalator Echoes", "artist": "Synth Wave Cadet", "duration_sec": 215.0},
    {"title": "Closed Mall Blues", "artist": "Coffee Shop Beats", "duration_sec": 165.0},
    {"title": "Security Desk Solitude", "artist": "Midnight Ranger", "duration_sec": 240.0},
    {"title": "After Hours Slumber", "artist": "Vaporwave Visionary", "duration_sec": 195.0}
]

FRAME_RATE = 30

def verify_master_audio_pipeline():
    print("=== RUNNING STREAM AUDIO INTEGRITY CHECKS ===")
    
    master_file = "lofi_music.mp3"
    
    # 1. Check if the compiled master loop file exists
    if not os.path.exists(master_file):
        print(f"[CRITICAL ERROR] '{master_file}' not found. Run your FFmpeg concat command first.")
        sys.exit(1)
        
    # 2. Extract technical headers from the master audio loop
    try:
        audio = MP3(master_file)
        actual_duration = audio.info.length
        sample_rate = audio.info.sample_rate
        bitrate = audio.info.bitrate
    except Exception as e:
        print(f"[CRITICAL ERROR] Failed to parse file header configurations. Details: {e}")
        sys.exit(1)
        
    # 3. Validate sample rate requirements
    if sample_rate != 44100:
        print(f"[WARNING] Sample rate is {sample_rate}Hz instead of 44100Hz. This can cause audio pitch distortion or latency drops on YouTube Live.")
    else:
        print("✅ Sample Rate Validated: 44,100 Hz")

    # 4. Calculate mathematical duration totals
    calculated_duration = sum([track["duration_sec"] for track in TRACKLIST])
    calculated_total_frames = sum([round(track["duration_sec"] * FRAME_RATE) for track in TRACKLIST])
    
    # 5. Measure precision sync drift thresholds
    drift_seconds = abs(actual_duration - calculated_duration)
    drift_frames = drift_seconds * FRAME_RATE
    
    print("\n--- MATRIX DATA MATCH ANALYSIS ---")
    print(f"Master File Real Duration: {actual_duration:.3f} seconds")
    print(f"Python Array Total Time:   {calculated_duration:.3f} seconds")
    print(f"Total Sequence Video Frames: {calculated_total_frames} frames")
    
    # Tolerable threshold error budget: less than 15 frames (half a second)
    if drift_frames > 15:
        print(f"\n[CRITICAL SYNC ERROR DETECTED]")
        print(f"⚠️ Your text tracker drifts from the real audio timeline by {drift_seconds:.3f} seconds ({drift_frames:.1f} frames).")
        print("This causes 'Now Playing' screen titles to drift out of sync over time.")
        print("Fix: Ensure all source tracks use Constant Bitrate (CBR) and recalculate via the analyzer.")
        sys.exit(1)
    else:
        print(f"✅ Timeline Sync Confirmed: Total script drift is negligible ({drift_frames:.2f} frames).")
        print(f"Calculated 24/7 Run Safety Profile: Excellent.")
        print("\n=== SYSTEM BROADCAST READY FOR LAUNCH ===")

if __name__ == "__main__":
    verify_master_audio_pipeline()
