The Master Optimized Python Engine (stream_engine.py)
import os
import sys
import csv
import math
import time
from datetime import datetime
import multiprocessing
from PIL import Image, ImageDraw, ImageFont, ImageEnhance

# --- 1. HARDWARE CORE OPTIMIZATIONS ---
import PIL
PIL.Image.set_num_threads(multiprocessing.cpu_count())

# --- 2. CONFIGURATION & CONSTANTS ---
FRAME_RATE = 30
TOTAL_FRAMES = 60 * 60 * 6 * FRAME_RATE  # Exactly 6 hours (648,000 frames)

# Tracklist Database Setup (Durations must match your MP3 exact seconds)
TRACKLIST = [
    {"title": "Late Night Patrol", "artist": "Lo-Fi Guard Collective", "duration_sec": 180},
    {"title": "Neon Escalator Echoes", "artist": "Synth Wave Cadet", "duration_sec": 215},
    {"title": "Closed Mall Blues", "artist": "Coffee Shop Beats", "duration_sec": 165},
    {"title": "Security Desk Solitude", "artist": "Midnight Ranger", "duration_sec": 240},
    {"title": "After Hours Slumber", "artist": "Vaporwave Visionary", "duration_sec": 195}
]

# --- 3. INFRASTRUCTURE & MAP GENERATORS ---
def build_track_frame_map(tracklist, fps):
    frame_map = []
    current_start = 0
    for track in tracklist:
        duration_frames = track["duration_sec"] * fps
        frame_map.append({
            "title": track["title"], "artist": track["artist"],
            "start_frame": current_start, "end_frame": current_start + duration_frames
        })
        current_start += duration_frames
    return frame_map, current_start

TRACK_FRAME_MAP, AUDIO_LOOP_DURATION = build_track_frame_map(TRACKLIST, FRAME_RATE)

# --- 4. PRE-FLIGHT ASSET VALDIATOR ---
def run_preflight_diagnostics():
    required = ["mall_structure_day.png", "mall_structure_night.png", 
                "guard_walk_right.png", "guard_walk_left.png", "mall_pillars.png"]
    for f in required:
        if not os.path.exists(f):
            print(f"[CRITICAL ERROR] Missing core asset: {f}", file=sys.stderr)
            sys.exit(1)
    if not os.path.exists("advertiser_log.csv"):
        # Auto-create empty spreadsheet log template if missing
        with open("advertiser_log.csv", "w", newline="") as f:
            writer = csv.writer(f)
            writer.writerow(["Storefront_ID", "Brand_Name", "Asset_File", "Start_Date", "End_Date", "Active_Status"])

run_preflight_diagnostics()

# --- 5. PERMANENT CACHED MEMORY LAYERS ---
bg_day = Image.open("mall_structure_day.png").convert("RGBA")
bg_night = Image.open("mall_structure_night.png").convert("RGBA")
guard_right = Image.open("guard_walk_right.png").convert("RGBA")
guard_left = Image.open("guard_walk_left.png").convert("RGBA")
foreground = Image.open("mall_pillars.png").convert("RGBA")

# Font Load Initialization
try:
    font_title = ImageFont.truetype("Arial.ttf", 28)
    font_artist = ImageFont.truetype("Arial.ttf", 20)
    font_dialogue = ImageFont.truetype("Arial.ttf", 22)
except IOError:
    font_title = font_artist = font_dialogue = ImageFont.load_default()

# --- 6. DYNAMIC LOOKUP LOGIC MODULES ---
def get_live_ads():
    # Set default fallback frames
    try: s1 = Image.open("fictional_store_1.png").convert("RGBA")
    except: s1 = Image.new("RGBA", (450, 300), (0, 0, 0, 0))
    try: s2 = Image.open("fictional_store_2.png").convert("RGBA")
    except: s2 = Image.new("RGBA", (450, 300), (0, 0, 0, 0))
    
    current_date = datetime.now().date()
    if not os.path.exists("advertiser_log.csv"): return s1, s2

    with open("advertiser_log.csv", mode="r") as file:
        for row in csv.DictReader(file):
            try:
                start_dt = datetime.strptime(row["Start_Date"], "%Y-%m-%d").date()
                end_dt = datetime.strptime(row["End_Date"], "%Y-%m-%d").date()
                if row["Active_Status"].strip().lower() == "true" and (start_dt <= current_date <= end_dt):
                    if os.path.exists(row["Asset_File"]):
                        ad_img = Image.open(row["Asset_File"]).convert("RGBA")
                        if row["Storefront_ID"] == "store_1": s1 = ad_img
                        elif row["Storefront_ID"] == "store_2": s2 = ad_img
            except: pass
    return s1, s2

def get_track_info(frame):
    normalized = frame % AUDIO_LOOP_DURATION
    for t in TRACK_FRAME_MAP:
        if t["start_frame"] <= normalized < t["end_frame"]:
            return t["title"], t["artist"], normalized - t["start_frame"]
    return "Ambient Track", "Mall Collective", 0

# --- 7. INFINITE PRODUCTION STREAM LOOP ---
def main():
    guard_x, guard_y, guard_speed, direction = 100, 500, 2, "right"
    frame = 0
    target_frame_time = 1.0 / FRAME_RATE # 0.0333 seconds per frame

    while True:
        start_time = time.time()
        cycle_frame = frame % TOTAL_FRAMES
        
        # A. Operations Shift / Closing Time Triggers
        is_open = cycle_frame <= (4 * 60 * 60 * FRAME_RATE) # Open first 4 hours
        canvas = bg_day.copy() if is_open else bg_night.copy()

        # B. Behavioral Security Guard Patrol State Machine
        if direction == "right":
            guard_x += guard_speed
            sprite = guard_right
            if guard_x > 1500: direction = "left"
        else:
            guard_x -= guard_speed
            sprite = guard_left
            if guard_x < 100: direction = "right"

        # C. Zero-Latency Layer Compositing
        ad_1, ad_2 = get_live_ads()
        canvas.alpha_composite(ad_1, dest=(200, 300)) # Store 1 Coordinates
        canvas.alpha_composite(ad_2, dest=(800, 300)) # Store 2 Coordinates
        canvas.alpha_composite(sprite, dest=(guard_x, guard_y))
        canvas.alpha_composite(foreground, dest=(0, 0))

        # D. Dynamic Music HUD & Dialogue Text Injection
        title, artist, song_frame = get_track_info(frame)
        draw = ImageDraw.Draw(canvas)
        draw.rectangle(, fill=(0, 0, 0, 140), radius=8)
        draw.text((60, 50), f"������ {title}", fill=(255, 255, 255), font=font_title)
        draw.text((60, 85), f"by {artist}", fill=(200, 200, 200), font=font_artist)

        # Draw Guard Welcome Dialogue for the first 4 seconds of any track
        if song_frame < (4 * FRAME_RATE):
            draw.rectangle([guard_x+20, guard_y-80, guard_x+250, guard_y-30], fill=(255, 255, 255, 230), radius=12)
            draw.text((guard_x+35, guard_y-68), "Loving this beat...", fill=(20, 20, 20), font=font_dialogue)

        # E. Chronological Environmental Color Grading
        time_factor = (math.sin(2 * math.pi * cycle_frame / TOTAL_FRAMES) + 1) / 2
        final_img = canvas.convert("RGB")
        final_img = ImageEnhance.Brightness(final_img).enhance(0.3 + (time_factor * 0.7))
        final_img = ImageEnhance.Color(final_img).enhance(0.5 + (time_factor * 0.5))

        if not is_open:
            final_img = Image.blend(final_img, Image.new("RGB", final_img.size, (5, 10, 35)), alpha=0.35)

        # F. Direct-To-RAM Pipe Streaming Execution
        try:
            # Writes raw compressed JPEG binary code straight into system standard out
            final_img.save(sys.stdout.buffer, "JPEG", quality=85)
        except IOError:
            # Breaks loop cleanly if FFmpeg shuts down or drops connection
            break

        # G. CPU/Hardware Thermal Regulator Sleep Timer
        elapsed_time = time.time() - start_time
        if elapsed_time < target_frame_time:
            time.sleep(target_frame_time - elapsed_time)
            
        frame += 1

if __name__ == "__main__":
    main()
The Terminal Stream Launcher Script (start_stream.sh)
Because the Python file now outputs raw frame binary structures directly through memory, update your Shell wrapper script to look exactly like this. Notice the python3 stream_engine.py | command passing the RAM stream directly into FFmpeg via the -i pipe:0 variable.
#!/bin/bash
STREAM_KEY="YOUR_YOUTUBE_STREAM_KEY"
AUDIO_FILE="lofi_music.mp3"
DISCORD_URL="https://discord.com"

send_alert() {
    curl -H "Content-Type: application/json" -X POST -d "{\"content\": \"$1\"}" "$DISCORD_URL"
}

send_alert "������ **Server Status Online**: Initializing Optimized RAM Data Pipeline Stream."

while true
do
    # Run python engine and pipe its standard RAM output directly into the FFmpeg input stream handler
    python3 stream_engine.py | ffmpeg -f image2pipe -framerate 30 -vcodec mjpeg -i pipe:0 \
    -stream_loop -1 -i "$AUDIO_FILE" \
    -c:v libx264 -pix_fmt yuv420p -preset veryfast -b:v 3000k -g 60 \
    -c:a aac -b:a 168k -ar 44100 -f flv "rtmp://://youtube.com"

    send_alert "⚠️ **STREAM REDIRECT WARNING**: Memory pipe broken or link dropped. Instantly rebooting system infrastructure..."
    sleep 5
done
