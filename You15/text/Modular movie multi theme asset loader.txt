Modular movie multi theme asset loader

Here is the multi-theme asset loader code. This engine calculates which 6-hour aesthetic block is currently active, handles unloading the old art assets from RAM to prevent memory bloat, and dynamically caches the new art style along with its matching advertiser logos on the fly.
The Multi-Theme Asset Loader Engine
Replace your previous asset initialization and loading sections with this modular object-oriented architecture:
import os
import csv
import sys
from datetime import datetime
from PIL import Image

# 1. THEME MATRIX CONFIGURATION
# 6 hours per block at 30 FPS = 648,000 frames per block
BLOCK_DURATION_FRAMES = 60 * 60 * 6 * 30 
TOTAL_DAILY_FRAMES = BLOCK_DURATION_FRAMES * 4 # 24 Hours total rotation

THEMES = {
    0: "pixel_art",    # 00:00 - 06:00
    1: "anime_vector", # 06:00 - 12:00
    2: "matte_painting",# 12:00 - 18:00
    3: "cyberpunk"     # 18:00 - 24:00
}

class ThemeAssetManager:
    def __init__(self):
        self.current_theme_id = None
        self.theme_name = ""
        
        # RAM Cache slots for active theme layers
        self.bg_day = None
        self.bg_night = None
        self.guard_right = None
        self.guard_left = None
        self.foreground = None
        self.sky_panorama = None
        self.flashlight_mask = None

    def load_theme_into_ram(self, theme_id):
        """Safely unloads old theme assets and caches new ones in RAM"""
        self.current_theme_id = theme_id
        self.theme_name = THEMES[theme_id]
        theme_path = f"assets/{self.theme_name}"
        
        print(f"\n[SYSTEM] >>> CHANGING NETWORK THEME TO: {self.theme_name.upper()} <<<", file=sys.stderr)
        
        # Clear previous images from memory explicitly to prevent RAM leakage
        self.bg_day = self.bg_night = self.guard_right = self.guard_left = None
        self.foreground = self.sky_panorama = self.flashlight_mask = None
        
        # Verify theme directory exists before loading
        if not os.path.exists(theme_path):
            print(f"[CRITICAL ERROR] Theme directory '{theme_path}' does not exist!", file=sys.stderr)
            sys.exit(1)
            
        try:
            # Load and cache the new 1080p layer structures natively in memory
            self.bg_day = Image.open(f"{theme_path}/mall_structure_day.png").convert("RGBA")
            self.bg_night = Image.open(f"{theme_path}/mall_structure_night.png").convert("RGBA")
            self.guard_right = Image.open(f"{theme_path}/guard_walk_right.png").convert("RGBA")
            self.guard_left = Image.open(f"{theme_path}/guard_walk_left.png").convert("RGBA")
            self.foreground = Image.open(f"{theme_path}/mall_pillars.png").convert("RGBA")
            self.sky_panorama = Image.open(f"{theme_path}/panoramic_sky.png").convert("RGBA")
            
            # Flashlight is optional depending on the style choice
            fl_path = f"{theme_path}/guard_flashlight_mask.png"
            if os.path.exists(fl_path):
                self.flashlight_mask = Image.open(fl_path).convert("RGBA")
            else:
                self.flashlight_mask = Image.new("RGBA", (1920, 1080), (0, 0, 0, 0))
                
            print(f"[SUCCESS] {self.theme_name.upper()} textures loaded into system memory seamlessly.", file=sys.stderr)
        except Exception as e:
            print(f"[CRITICAL ERROR] Failed to load theme {self.theme_name}. Details: {e}", file=sys.stderr)
            sys.exit(1)

    def get_theme_advertisements(self):
        """Fetches active ads that specifically match the current time-block theme style"""
        # Universal transparent fallbacks
        s1 = Image.new("RGBA", (450, 350), (0, 0, 0, 0))
        s2 = Image.new("RGBA", (450, 350), (0, 0, 0, 0))
        
        current_date = datetime.now().date()
        if not os.path.exists("advertiser_log.csv"): 
            return s1, s2

        with open("advertiser_log.csv", mode="r") as file:
            for row in csv.DictReader(file):
                try:
                    start_dt = datetime.strptime(row["Start_Date"], "%Y-%m-%d").date()
                    end_dt = datetime.strptime(row["End_Date"], "%Y-%m-%d").date()
                    is_active = row["Active_Status"].strip().lower() == "true"
                    
                    # Target specific rows matching date, active status, AND the current visual theme
                    if is_active and (start_dt <= current_date <= end_dt) and row["Theme_Style"] == self.theme_name:
                        if os.path.exists(row["Asset_File"]):
                            ad_img = Image.open(row["Asset_File"]).convert("RGBA")
                            if row["Storefront_ID"] == "store_1": 
                                s1 = ad_img
                            elif row["Storefront_ID"] == "store_2": 
                                s2 = ad_img
                except Exception:
                    pass
        return s1, s2

# Initialize the theme engine before entering the infinite loop
theme_manager = ThemeAssetManager()

frame = 0
while True:
    # 1. TIMELINE CALCULATIONS
    # Track position within the full 24-hour daily cycle
    daily_cycle_frame = frame % TOTAL_DAILY_FRAMES
    
    # Track position within the current active 6-hour aesthetic block
    current_block_frame = daily_cycle_frame % BLOCK_DURATION_FRAMES
    
    # Calculate target theme block integer identifier (0, 1, 2, or 3)
    target_theme_id = daily_cycle_frame // BLOCK_DURATION_FRAMES
    
    # 2. SEAMLESS RE-CAPPING MEMORY CHECK
    # Trigger asset load if starting the stream or crossing the 6-hour threshold boundary
    if theme_manager.current_theme_id != target_theme_id:
        theme_manager.load_theme_into_ram(target_theme_id)
        
    # 3. COMPOSITION ENGINE UPDATE
    # Update all your layer assignments to pull directly from the theme manager cache
    is_open = current_block_frame <= (4 * 60 * 60 * 30) # Open first 4 hours of the block
    
    if is_open:
        frame_canvas = theme_manager.bg_day.copy()
    else:
        frame_canvas = theme_manager.bg_night.copy()
