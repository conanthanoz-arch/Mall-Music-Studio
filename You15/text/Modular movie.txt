import os
import sys
import math
import time
import multiprocessing
from PIL import Image, ImageDraw, ImageEnhance

# --- 1. MULTI-CORE HARDWARE OPTIMIZATION ---
import PIL
PIL.Image.set_num_threads(multiprocessing.cpu_count())

# --- 2. CORE BROADCAST SETTINGS ---
FRAME_RATE = 30
# 6 Hours per visual block (Pixel Art -> Anime Vector)
BLOCK_FRAMES = 60 * 60 * 6 * FRAME_RATE 
TOTAL_CYCLE_FRAMES = BLOCK_FRAMES * 2  # 12-Hour total rotation loop

# --- 3. HARDWARE PRE-FLIGHT DIRECTORY GATE ---
def initialize_system_directories():
    required_paths = [
        "assets/pixel_art",
        "assets/anime_vector"
    ]
    for path in required_paths:
        if not os.path.exists(path):
            print(f"[ERROR] Missing structural path: '{path}'", file=sys.stderr)
            sys.exit(1)
    print("[SUCCESS] Merry Mall folder infrastructure validated.", file=sys.stderr)

initialize_system_directories()

# --- 4. PRE-LOADING ASSETS DIRECTLY TO SYSTEM RAM ---
# (For your initial test loop, load placeholder images matching your real-world dimensions)
try:
    print("[SYSTEM] Caching aesthetic assets into RAM...", file=sys.stderr)
    bg_day = Image.open("assets/pixel_art/mall_structure_day.png").convert("RGBA")
    bg_night = Image.open("assets/pixel_art/mall_structure_night.png").convert("RGBA")
    guard_right = Image.open("assets/pixel_art/guard_walk_right.png").convert("RGBA")
    guard_left = Image.open("assets/pixel_art/guard_walk_left.png").convert("RGBA")
    foreground = Image.open("assets/pixel_art/mall_pillars.png").convert("RGBA")
except Exception as e:
    print(f"[CRITICAL ERROR] Failed to cache base assets into memory. Details: {e}", file=sys.stderr)
    sys.exit(1)

# --- 5. INITIAL OPERATIONAL VARIABLES ---
# Locked to your real-world 1080p human grid metrics
guard_x = 100
guard_y = 675  # Feet rest flat at Y: 850
guard_speed = 3
direction = "right"

target_frame_time = 1.0 / FRAME_RATE
frame = 0

print("\n������ MERRY MALL CORE INITIALIZED. STREAM PIPE OPEN.\n", file=sys.stderr)

# --- 6. INFINITE BROADCAST PIPELINE LOOP ---
while True:
    start_time = time.time()
    
    # Track position within the current visual block loop
    cycle_frame = frame % TOTAL_CYCLE_FRAMES
    current_block_frame = cycle_frame % BLOCK_FRAMES
    
    # Determine the status of mall operations (Open for first 4 hours of the 6-hour block)
    is_mall_open = current_block_frame <= (4 * 60 * 60 * FRAME_RATE)

    # --- PHASE A: PATROL BEHAVIOR LOGIC ---
    if direction == "right":
        guard_x += guard_speed
        current_sprite = guard_right
        if guard_x > 1750:  # Canvas boundary width block check
            direction = "left"
    else:
        guard_x -= guard_speed
        current_sprite = guard_left
        if guard_x < 100:
            direction = "right"

    # --- PHASE B: LAYER COMPOSITING STACK ---
    # Draw base architectural wall layers from memory cache
    frame_canvas = bg_day.copy() if is_mall_open else bg_night.copy()
    
    # Overlap the guard character over the background
    frame_canvas.alpha_composite(current_sprite, dest=(guard_x, guard_y))
    
    # Layer the foreground structural safety columns over the guard
    frame_canvas.alpha_composite(foreground, dest=(0, 0))

    # --- PHASE C: COLOR TEMPERATURE FILTERS ---
    # Calculate a smooth sunset/sunrise wave curve
    time_factor = (math.sin(2 * math.pi * current_block_frame / BLOCK_FRAMES) + 1) / 2
    
    final_frame = frame_canvas.convert("RGB")
    final_frame = ImageEnhance.Brightness(final_frame).enhance(0.4 + (time_factor * 0.6))
    final_frame = ImageEnhance.Color(final_frame).enhance(0.6 + (time_factor * 0.4))

    # Apply specialized midnight blue tint matrix if mall lights cut shut
    if not is_mall_open:
        night_overlay = Image.new("RGB", final_frame.size, (10, 15, 45))
        final_frame = Image.blend(final_frame, night_overlay, alpha=0.30)

    # --- PHASE D: DIRECT-TO-RAM ZERO-LATENCY OUT PIPE ---
    try:
        # Save raw frame bytes directly to your computer's standard output binary stream
        final_frame.save(sys.stdout.buffer, "JPEG", quality=85)
    except IOError:
        # Breaks cleanly if the downstream FFmpeg media server link drops out
        print("[WARNING] Downstream streaming pipe broken. Stopping core loop.", file=sys.stderr)
        break

    # --- PHASE E: HARDWARE THERMAL REGULATOR CLOCK ---
    elapsed_time = time.time() - start_time
    if elapsed_time < target_frame_time:
        time.sleep(target_frame_time - elapsed_time)

    frame += 1
